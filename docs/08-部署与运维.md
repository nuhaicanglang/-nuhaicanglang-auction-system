## 一、环境总览

### 1.1 三套环境

| 环境 | 用途 | 配置 |
|------|-----|------|
| dev   | 本机开发 | 各组件 Docker 起,Spring Boot IDE 启动 |
| test  | 联调/演示 | 整套 docker-compose,模拟生产 |
| prod  | 上线(选做) | 云服务器,域名 + HTTPS |

### 1.2 系统要求

- Windows 11 / Linux Ubuntu 22.04 / macOS 都可
- 内存 ≥ 8GB(ES 比较占)
- 磁盘 ≥ 30GB
- 已安装 Docker Desktop、Git、JDK 17、Node 20、Maven 3.9

## 三、开发环境(中间件)

### 3.1 `docker-compose.middleware.yml`

只起中间件,后端用 IDE 直接 run/debug。

```yaml
version: "3.8"

networks:
  auction-net:
    driver: bridge

services:
  mysql:
    image: mysql:8.0
    container_name: auction-mysql
    restart: unless-stopped
    environment:
      MYSQL_ROOT_PASSWORD: ${MYSQL_ROOT_PASSWORD:-root123456}
      MYSQL_DATABASE: auction
      TZ: Asia/Shanghai
    ports:
      - "3306:3306"
    volumes:
      - ./mysql/data:/var/lib/mysql
      - ./mysql/init.sql:/docker-entrypoint-initdb.d/01-init.sql:ro
      - ./mysql/my.cnf:/etc/mysql/conf.d/my.cnf:ro
    command:
      - --default-authentication-plugin=mysql_native_password
      - --character-set-server=utf8mb4
      - --collation-server=utf8mb4_0900_ai_ci
    networks: [auction-net]

  redis:
    image: redis:7.2-alpine
    container_name: auction-redis
    restart: unless-stopped
    ports:
      - "6379:6379"
    volumes:
      - ./redis/data:/data
      - ./redis/redis.conf:/etc/redis/redis.conf:ro
    command: redis-server /etc/redis/redis.conf
    networks: [auction-net]

  rabbitmq:
    image: rabbitmq:3.12-management
    container_name: auction-rabbitmq
    restart: unless-stopped
    environment:
      RABBITMQ_DEFAULT_USER: ${RABBITMQ_USER:-admin}
      RABBITMQ_DEFAULT_PASS: ${RABBITMQ_PASS:-admin123}
    ports:
      - "5672:5672"     # AMQP
      - "15672:15672"   # Management UI
    volumes:
      - ./rabbitmq/data:/var/lib/rabbitmq
    networks: [auction-net]

  elasticsearch:
    image: docker.elastic.co/elasticsearch/elasticsearch:8.11.3
    container_name: auction-es
    restart: unless-stopped
    environment:
      - discovery.type=single-node
      - xpack.security.enabled=false
      - "ES_JAVA_OPTS=-Xms512m -Xmx512m"
    ports:
      - "9200:9200"
    volumes:
      - ./elasticsearch/data:/usr/share/elasticsearch/data
      - ./elasticsearch/plugins:/usr/share/elasticsearch/plugins
    networks: [auction-net]

  kibana:
    image: docker.elastic.co/kibana/kibana:8.11.3
    container_name: auction-kibana
    restart: unless-stopped
    environment:
      - ELASTICSEARCH_HOSTS=http://elasticsearch:9200
      - I18N_LOCALE=zh-CN
    ports:
      - "5601:5601"
    depends_on: [elasticsearch]
    networks: [auction-net]

  minio:
    image: minio/minio:latest
    container_name: auction-minio
    restart: unless-stopped
    environment:
      MINIO_ROOT_USER: ${MINIO_USER:-admin}
      MINIO_ROOT_PASSWORD: ${MINIO_PASS:-admin123456}
    ports:
      - "9000:9000"     # API
      - "9001:9001"     # Console
    volumes:
      - ./minio/data:/data
    command: server /data --console-address ":9001"
    networks: [auction-net]
```

## 六、生产编排(`docker-compose.yml`)

包含中间件 + 后端 + 前端 + Nginx,一键起整套。

```yaml
version: "3.8"
networks:
  auction-net:
    driver: bridge

services:
  mysql:           # 同上
  redis:           # 同上
  rabbitmq:        # 同上
  elasticsearch:   # 同上

  backend:
    build:
      context: ../auction-backend
      dockerfile: Dockerfile
    image: auction/backend:latest
    container_name: auction-backend
    restart: unless-stopped
    environment:
      SPRING_PROFILES_ACTIVE: prod
      MYSQL_HOST: mysql
      MYSQL_PASSWORD: ${MYSQL_ROOT_PASSWORD}
      REDIS_HOST: redis
      RABBITMQ_HOST: rabbitmq
      ES_HOST: elasticsearch
      JWT_SECRET: ${JWT_SECRET}
    ports:
      - "8080:8080"
    depends_on:
      - mysql
      - redis
      - rabbitmq
      - elasticsearch
    volumes:
      - ./backend/logs:/app/logs
    networks: [auction-net]

  frontend:
    build:
      context: ../auction-frontend
      dockerfile: Dockerfile
    image: auction/frontend:latest
    container_name: auction-frontend
    restart: unless-stopped
    networks: [auction-net]

  nginx:
    image: nginx:1.25-alpine
    container_name: auction-nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/conf.d:/etc/nginx/conf.d:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro
      - ./nginx/logs:/var/log/nginx
    depends_on: [backend, frontend]
    networks: [auction-net]
```

## 七、后端 Dockerfile

```dockerfile
# 多阶段构建,减小镜像
FROM eclipse-temurin:17-jdk-alpine AS builder
WORKDIR /build
COPY pom.xml ./
COPY auction-common/pom.xml auction-common/
COPY auction-framework/pom.xml auction-framework/

COPY . .
RUN mvn clean package -DskipTests -B

FROM eclipse-temurin:17-jre-alpine
WORKDIR /app
RUN addgroup -S app && adduser -S app -G app
COPY --from=builder /build/auction-admin/target/auction-admin-*.jar app.jar
RUN chown -R app:app /app
USER app
EXPOSE 8080
HEALTHCHECK --interval=30s --timeout=10s --retries=3 \
  CMD wget --quiet --tries=1 --spider http://localhost:8080/actuator/health || exit 1
ENTRYPOINT ["sh","-c","java $JAVA_OPTS -jar app.jar --spring.profiles.active=$SPRING_PROFILES_ACTIVE"]
```

## 九、Nginx 主配置(`nginx/conf.d/auction.conf`)

```nginx
upstream backend {
    server backend:8080;
    keepalive 32;
}

upstream frontend {
    server frontend:80;
}

# HTTP → HTTPS(prod)
server {
    listen 80;
    server_name auction.example.com;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name auction.example.com;

    ssl_certificate     /etc/nginx/ssl/cert.pem;
    ssl_certificate_key /etc/nginx/ssl/key.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # Gzip
    gzip on;
    gzip_min_length 1k;
    gzip_types text/css text/plain application/javascript application/json image/svg+xml;
    gzip_vary on;

    # 限流
    limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;

    client_max_body_size 30M;

    # 静态资源
    location / {
        proxy_pass http://frontend;
        proxy_set_header Host $host;
    }

    location ~* \.(js|css|png|jpg|jpeg|gif|webp|woff2)$ {
        proxy_pass http://frontend;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }

    # API
    location /api/ {
        limit_req zone=api burst=20 nodelay;
        proxy_pass http://backend;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 30s;
    }

    # WebSocket
    location /api/v1/ws {
        proxy_pass http://backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade           $http_upgrade;
        proxy_set_header Connection        "upgrade";
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_read_timeout 3600s;
        proxy_send_timeout 3600s;
    }
}
```

## 十、一键部署脚本

### 10.1 `scripts/build-backend.sh`

```bash
#!/bin/bash
set -e
cd "$(dirname "$0")/../auction-backend"
mvn clean package -DskipTests -B
echo "[backend] build success: auction-admin/target/auction-admin-*.jar"
```

### 10.2 `scripts/build-frontend.sh`

```bash
#!/bin/bash
set -e
cd "$(dirname "$0")/../auction-frontend"
npm ci
npm run build
echo "[frontend] build success: dist/"
```

### 10.3 `scripts/deploy-prod.sh`

```bash
#!/bin/bash
set -e
cd "$(dirname "$0")/.."

echo ">>> 1. 拉取最新代码"
git pull

echo ">>> 2. 构建后端"
./scripts/build-backend.sh

echo ">>> 3. 构建前端"
./scripts/build-frontend.sh

echo ">>> 4. 重建镜像并重启"
cd deploy
docker compose build backend frontend
docker compose up -d --no-deps backend frontend
docker compose exec nginx nginx -s reload

echo ">>> 5. 健康检查"
sleep 5
curl -fsS http://localhost:8080/actuator/health || (echo "健康检查失败" && exit 1)
echo "[deploy] success"
```

## 十一、备份与恢复

### 11.1 MySQL 全量备份(`deploy/backup/mysql-backup.sh`)

```bash
#!/bin/bash
set -e
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR=/var/backups/auction/mysql
mkdir -p $BACKUP_DIR

docker exec auction-mysql sh -c \
  "exec mysqldump -uroot -p\"$MYSQL_ROOT_PASSWORD\" --single-transaction --routines --triggers --master-data=2 auction" \
  | gzip > $BACKUP_DIR/auction_$DATE.sql.gz

# 保留 14 天
find $BACKUP_DIR -name "auction_*.sql.gz" -mtime +14 -delete
```

加 cron:

```bash
0 2 * * * /path/to/mysql-backup.sh >> /var/log/auction-backup.log 2>&1
```

## 十四、CI/CD(选做加分)

GitHub Actions 流水线 `.github/workflows/ci.yml`:

```yaml
name: CI

on:
  push:
    branches: [main, develop]
  pull_request:

jobs:
  build-backend:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-java@v4
        with:
          distribution: temurin
          java-version: 17
      - name: Build
        run: |
          cd auction-backend
          mvn clean verify -B

  build-frontend:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
      - name: Build
        run: |
          cd auction-frontend
          npm ci
          npm run build
```
